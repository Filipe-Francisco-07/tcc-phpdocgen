<?php

namespace Generator;

/**
 * Classe responsável por aplicar documentação em conteúdos de texto, substituindo placeholders por docblocks correspondentes.
 */
final class AplicadorDocumentacao
{
    /**
     * Processa o conteúdo fornecido, substituindo placeholders por documentações correspondentes.
     * 
     * @param string $sConteudo O conteúdo que contém os placeholders para substituição.
     * @param array $aDocs Um array associativo que mapeia IDs de documentação para suas respectivas strings.
     * @return string O conteúdo resultante após a substituição dos placeholders.
     */
    public function aplicar(string $sConteudo, array $aDocs): string
    {
        $sConteudo = str_replace("\r\n", "\n", $sConteudo);
        $aLinhas   = explode("\n", $sConteudo);

        $iTotal = count($aLinhas);
        for ($i = 0; $i < $iTotal; $i++) {
            $sLinha = $aLinhas[$i];

            if (!preg_match('/^\s*\{\{\s*(doc_[A-Za-z0-9_-]+)\s*\}\}\s*$/', $sLinha, $m)) {
                continue;
            }
            $sId = $m[1];
            if (!isset($aDocs[$sId])) {
                continue;
            }

            // identaÃ§Ã£o da prÃ³pria linha do placeholder
            preg_match('/^([ \t]*)/', $sLinha, $mi);
            $sIndent = $mi[1] ?? '';

            // fallback: identaÃ§Ã£o da prÃ³xima linha Ãºtil
            if ($sIndent === '' && $i + 1 < $iTotal) {
                $j = $i + 1;
                while ($j < $iTotal && trim($aLinhas[$j]) === '') {
                    $j++;
                }
                if ($j < $iTotal && preg_match('/^([ \t]+)/', $aLinhas[$j], $mn)) {
                    $sIndent = $mn[1];
                }
            }

            $aLinhas[$i] = $this->paraDocblockComIdentacao($aDocs[$sId], $sIndent);
        }

        return implode("\n", $aLinhas);
    }

    /**
     * Formata um texto em um bloco de documentação com indentação.
     * 
     * O texto é processado para garantir que esteja no formato correto de um DocBlock, 
     * adicionando indentação conforme necessário.
     * 
     * @param string $sTexto O texto a ser formatado como DocBlock.
     * @param string $sIndent A string de indentação a ser aplicada a cada linha do DocBlock.
     * @return string O texto formatado como um DocBlock com a indentação especificada.
     */
    private function paraDocblockComIdentacao(string $sTexto, string $sIndent = ''): string
    {
        $sTexto = trim(str_replace("\r\n", "\n", $sTexto));

        if (!str_starts_with($sTexto, '/**')) {
            $aLinhas = $sTexto === '' ? ['DocumentaÃ§Ã£o gerada.'] : explode("\n", $sTexto);
            $aLinhas = array_map(fn ($l) => ' * ' . ltrim(preg_replace('/^\*\s*/', '', $l)), $aLinhas);
            $sTexto  = "/**\n" . implode("\n", $aLinhas) . "\n */";
        }

        $aOut = array_map(fn ($l) => $sIndent . $l, explode("\n", $sTexto));
        return implode("\n", $aOut);
    }
}
